                    }

                    }

                        return !this.options.isMobile;
                    },
                    appendTo: ".flipbook-menu",
                    title   : "Үлкейту",
                    "class" : "fa-plus",
                    template: "<div></div>",
                    callback: function (e) {
                        this.zoomIn();
                    }

                    enabled : function () {
                        return !this.options.isMobile;

                    appendTo: ".flipbook-menu",
                    title   : "Кішірейту",
                    "class" : "fa-minus",
                    template: "<div></div>",
                    callback: function (e) {
                        this.zoomOut();
                    }
                },

                        this.bookmarkShowing && this["btnbookmarks"] && this["btnbookmarks"].removeClass("active"),
                        this.thumbsCreated && this["btnpages"] && this["btnpages"].removeClass("active"),
                                $(e.target).toggleClass("active"), this.toggleToc();
                    }
                },
                pages: {
                    enabled: function() {
                        return !this.options.isMobile;
                    },
                    appendTo: ".flipbook-menu",
                    title: "Алдын ала қарау",
                    "class": "fa-th-large",
                    template: "<div></div>",
                    callback: function(e) {
                        this.tocShowing && this["btncontent"] && this["btncontent"].removeClass("active"),
                        this.bookmarkShowing && this["btnbookmarks"] && this["btnbookmarks"].removeClass("active"),
                                $(e.target).toggleClass("active"), this.toggleThumbs();
                    }
                },
                bookmarks: {
                    enabled: true,
                    appendTo: ".flipbook-menu",
                    title: "Бетбелгілер",
                    "class": "fa-bookmark",
                    template: "<div></div>",
                    callback: function(e) {
                        this.tocShowing && this["btncontent"] && this["btncontent"].removeClass("active"),
                        this.thumbsCreated && this["btnpages"] && this["btnpages"].removeClass("active"),
                                $(e.target).toggleClass("active"), this.toggleBookmarksList();
                    }
                },
                bookmark      : {
                    enabled : true,
                    prepend : ".flipbook-currentPageInner",
                    title   : "Бетбелгілерге сақтау",
                    "class" : "fa-bookmark-o",
                    template: "<div></div>",
                    callback: function (e) {
                        this.toggleBookmark();
                    }
                },
                info          : {

                    appendTo: ".flipbook-menu",
                    title   : "Атауын бекіту",
                    "class" : "fa-info",
                    template: "<div></div>",
                    callback: function (e) {
                        $(e.target).toggleClass("active"), this.infoFreeze = !this.infoFreeze, this.infoBarToggle()
                    }
                },
                sound         : {
                    enabled : true,
                    appendTo: ".flipbook-menu",
                    title   : "Дауысын қосу/өшіру",
                    "class" : "fa-volume-up",
                    template: "<div></div>",
                    callback: function (e) {
                        this.options.sound ?
                                (this.options.sound = false, $(e.target).addClass("fa-volume-off").removeClass("fa-volume-up")) :
                                (this.options.sound = true, $(e.target).addClass("fa-volume-up").removeClass("fa-volume-off"))
                    }
                },
                fullscreen    : {
                    enabled : function () {
                        return !this.options.isMobile;
                    },
                    appendTo: ".flipbook-menu",
                    title   : "Толық экран",
                    "class" : "fa-expand",
                    template: "<div></div>",
                    callback: function (e) {
                        this.elem.hasClass("flipbook-browser-fullscreen") ?
                                $(e.target).addClass("fa-expand").removeClass("fa-compress") :
                                $(e.target).addClass("fa-compress").removeClass("fa-expand")
                        this.toggleExpand();
                    }
                }
            }
